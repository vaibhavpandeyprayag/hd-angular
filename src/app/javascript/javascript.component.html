JavaScript

let keyword

Both the increment operations are done by declaring variables of same name in different scopes.

               
    increaseX() {
        {
            let x = this.letX.xBy1;
            x++;
            this.letX.xBy1 = x;
        }
        {
            let x = this.letX.xBy2;
            x += 2;
            this.letX.xBy2 = x;
        }
    }

const keyword

First x can be increased as it is declared using let keyword, but the second one declared using const keyword. Therefore, increasing second one raises error.

               
    increaseConstX()  {
        {
            let x = this.constX.xLet;
            x++;
            this.constX.xLet = x;
        }
        {
            const x = this.constX.xConst;
            // x++ => gives error
        }
    }

for/of Loop

Digits: {{digitsString}}

Used for/of loop to append digits to a string

               
    digits = [1, 2, 3, 4, 5, 6, 7, 8, 9, 0];
    digitsString = "";
    digitsToString() {
        for (let d of this.digits) this.digitsString += d;
        return this.digitsString;
    }

Map

Used map to store key-value pairs and used Select input to choose key.

               
    map = new Map([["fruit", "apple"],["shape", "cube"],["vegetable", "tomato"], ["company", "humanitics"], ["place", "noida"]]);
    valueSelected = this.map.get("fruit");
    showValue(event: Event) {
        let target = event.target as HTMLSelectElement;
        this.valueSelected = this.map.get(target.value);
    }
